13  package biz.gabrys.maven.plugins.directory.content.file;
14  
15  import java.io.File;
16  
17  /**
18   * Stores {@link FileMetadataFactory} configuration.
19   * @since 1.0
20   */
21  public class FileMetadataFactoryConfiguration {
22  
23      private File rootDirectory;
24      private String separator;
25  
26      /**
27       * Returns root directory for files paths.
28       * @return the root directory.
29       * @since 1.0
30       */
31      public File getRootDirectory() {
32          return rootDirectory;
33      }
34  
35      /**
36       * Sets root directory for files paths.
37       * @param rootDirectory the root directory.
38       * @since 1.0
39       */
40      public void setRootDirectory(final File rootDirectory) {
41          this.rootDirectory = rootDirectory;
42      }
43  
44      /**
45       * Returns file path separator.
46       * @return the path separator.
47       * @since 1.0
48       */
49      public String getSeparator() {
50          return separator;
51      }
52  
53      /**
54       * Sets file path separator.
55       * @param separator the separator.
56       * @since 1.0
57       * @see File#separator
58       */
59      public void setSeparator(final String separator) {
60          this.separator = separator;
61      }
62  }
