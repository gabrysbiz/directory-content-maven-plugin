13  package biz.gabrys.maven.plugins.directory.content.file;
14  
15  import java.io.File;
16  
17  /**
18   * Responsible for getting parent directory path for {@link File} objects.
19   * @since 1.0
20   */
21  public class DirectoryResolver {
22  
23      /**
24       * Returns parent directory path for file.
25       * @param file the file.
26       * @param rootDirectory the root directory for file path.
27       * @param separator the file path separator used in returned path.
28       * @return the parent directory path (never {@code null}).
29       * @since 1.0
30       */
31      public String resolve(final File file, final File rootDirectory, final String separator) {
32          final int index = rootDirectory.getAbsolutePath().length();
33          String path = file.getParentFile().getAbsolutePath().substring(index);
34          if (path.startsWith(File.separator)) {
35              path = path.substring(File.separator.length());
36          }
37          if (File.separator.equals(separator)) {
38              return path;
39          }
40          return path.replace(File.separator, separator);
41      }
42  }
